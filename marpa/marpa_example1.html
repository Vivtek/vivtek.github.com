:default ::= action => [name,values]
lexeme default = latm => 1
:start ::= line

line ::= name_group sep parmgroup
name_group ::= tag | tag whitespace names

sep ::= whitespace
sep ::=
whitespace ~ [\s]+

tag ::= tag_name
tag_name ~  | 
 ~ [\w]+
 ~ '['  ']'

names ::= name+
name ~ [\w]+

parmgroup ::= paren_group
parmgroup ::=
paren_group ::= '(' parm_list ')'
parm_list ::= parm* separator => 
parm ::= words | words '=' words
words ::= name+

 ~ [,]
